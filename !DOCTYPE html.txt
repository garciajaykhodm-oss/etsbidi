    <script>
        // 🎯 CUSTOMIZE HERE:
        const HIS_NAME = "Alex"; // Change to his name
        const BIRTHDAY = new Date(2024, 9, 15); // His birthday (YYYY, MM-1, DD)
        
        // 🎵 YOUR CUSTOM MUSIC OPTIONS:
        const MUSIC_OPTIONS = {
            romantic: "https://www.soundjay.com/misc/sounds/romantic-piano.mp3", // Replace with your romantic song
            party: "https://www.soundjay.com/misc/sounds/party-horns.mp3", // Replace with party song
            custom: "https://your-song-link-here.com/your-love-song.mp3" // YOUR PERSONAL SONG!
        };
        
        let isPlaying = false;
        const bgMusic = document.getElementById('bgMusic');
        let currentMusic = 'romantic';
        
        // REVEAL SURPRISE FUNCTION
        function revealSurprise() {
            // Hide welcome screen
            document.getElementById('welcomeScreen').style.display = 'none';
            
            // Show main content
            document.getElementById('mainContainer').style.display = 'block';
            document.querySelector('.music-controls').style.display = 'block';
            
            // START MUSIC AUTOMATICALLY
            playMusic();
            
            // Start countdown
            setInterval(updateCountdown, 1000);
            updateCountdown();
        }
        
        // MUSIC FUNCTIONS
        function playMusic() {
            bgMusic.src = MUSIC_OPTIONS[currentMusic];
            bgMusic.play().catch(e => console.log('Autoplay prevented'));
            isPlaying = true;
        }
        
        function toggleMusic() {
            if (isPlaying) {
                bgMusic.pause();
                isPlaying = false;
            } else {
                bgMusic.play();
                isPlaying = true;
            }
        }
        
        function changeMusic(type) {
            currentMusic = type;
            if (isPlaying) {
                playMusic();
            }
        }
        
        // COUNTDOWN FUNCTION (same as before)
        function updateCountdown() {
            const now = new Date().getTime();
            const distance = BIRTHDAY - now;
            
            if (distance < 0) {
                document.getElementById('countdown').innerHTML = `
                    <div style="font-size: 2rem; color: #ff6b6b; font-weight: 700;">
                        HAPPY BIRTHDAY LOVER! 🎉🥳💖
                    </div>
                `;
                createFireworks();
                return;
            }
            
            const days = Math.floor(distance / (1000 * 60 * 60 * 24));
            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((distance % (1000 * 60)) / 1000);
            
            document.getElementById('countdown').innerHTML = `
                <div class="time-unit"><span class="time-number">${days}</span><span class="time-label">Days</span></div>
                <div class="time-unit"><span class="time-number">${hours}</span><span class="time-label">Hours</span></div>
                <div class="time-unit"><span class="time-number">${minutes}</span><span class="time-label">Minutes</span></div>
                <div class="time-unit"><span class="time-number">${seconds}</span><span class="time-label">Seconds</span></div>
            `;
            
            // Update title with his name
            document.querySelector('h1').textContent = `Happy 21st Birthday, ${HIS_NAME}! 🎂✨`;
            document.querySelector('.message').innerHTML = `
                ${HIS_NAME}, my love! 🥰<br><br>
                Turning 21 with you is my dream come true. You're my favorite person, 
                my best adventure, and the reason I smile every day. 
                I love you more than words can say! 💕<br><br>
                Here's to US and a million more birthdays together! 🥂
            `;
        }
        
        // FIREWORKS (same as before)
        function createFireworks() {
            const fireworks = document.getElementById('fireworks');
            for (let i = 0; i < 50; i++) {
                setTimeout(() => {
                    const firework = document.createElement('div');
                    firework.style.cssText = `
                        position: absolute;
                        left: ${Math.random() * 100}%;
                        top: ${Math.random() * 100}%;
                        width: 6px; height: 6px;
                        background: hsl(${Math.random() * 360}, 100%, 50%);
                        border-radius: 50%;
                        pointer-events: none;
                        animation: fireworkAnim 2s ease-out forwards;
                        z-index: 100;
                    `;
                    fireworks.appendChild(firework);
                    setTimeout(() => firework.remove(), 2000);
                }, i * 80);
            }
        }
        
        const style = document.createElement('style');
        style.textContent = `@keyframes fireworkAnim {
            0% { transform: scale(0) translateY(0); opacity: 1; }
            100% { transform: scale(1) translateY(-100vh); opacity: 0; }
        }`;
        document.head.appendChild(style);
        
        // Prevent right-click & make whole page clickable
        document.addEventListener('contextmenu', e => e.preventDefault());
        document.body.addEventListener('click', () => {
            if (document.getElementById('welcomeScreen').style.display !== 'none') {
                revealSurprise();
            }
        });
    </script>
