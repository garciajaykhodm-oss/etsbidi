    <script>
        /* 🎂 MATTY'S 21st BIRTHDAY: April 21, 2005 → 2026 */
        const BIRTHDAY = new Date(2026, 3, 21); // ✅ APRIL 21st SET!
        
        const MUSIC = {
            libu: "https://www.soundjay.com/misc/sounds/romantic-piano.mp3",
            party: "https://www.soundjay.com/misc/sounds/party-horns.mp3"
        };
        
        let isPlaying = false; const bgMusic = document.getElementById('bgMusic');
        
        function revealSurprise() {
            document.getElementById('welcomeScreen').style.display = 'none';
            document.getElementById('mainContainer').style.display = 'block';
            document.querySelector('.music-controls').style.display = 'block';
            playMusic(); setInterval(updateCountdown, 1000); updateCountdown();
        }
        
        function playMusic() { 
            bgMusic.src = MUSIC.libu; 
            bgMusic.play().catch(() => {}); 
            isPlaying = true; 
        }
        function toggleMusic() { isPlaying ? bgMusic.pause() : bgMusic.play(); isPlaying = !isPlaying; }
        function changeMusic(type) { if(MUSIC[type]) { bgMusic.src = MUSIC[type]; if(isPlaying) bgMusic.play(); } }
        
        function updateCountdown() {
            const distance = BIRTHDAY - new Date();
            if(distance < 0) {
                document.getElementById('countdown').innerHTML = '<div style="font-size:2rem;color:#fff;font-weight:700;">HAPPY BIRTHDAY BEBU! 🎉💖</div>';
                createFireworks(); return;
            }
            const d = Math.floor(distance / (1000*60*60*24));
            const h = Math.floor((distance%(1000*60*60*24))/(1000*60*60));
            const m = Math.floor((distance%(1000*60*60))/(1000*60));
            const s = Math.floor((distance%(1000*60))/1000);
            document.getElementById('countdown').innerHTML = `
                <div class="time-unit"><span class="time-number">${d}</span><span class="time-label">Days</span></div>
                <div class="time-unit"><span class="time-number">${h}</span><span class="time-label">Hours</span></div>
                <div class="time-unit"><span class="time-number">${m}</span><span class="time-label">Minutes</span></div>
                <div class="time-unit"><span class="time-number">${s}</span><span class="time-label">Seconds</span></div>
            `;
        }
        
        function createFireworks() {
            const fw = document.getElementById('fireworks');
            for(let i = 0; i < 50; i++) setTimeout(() => {
                const spark = document.createElement('div');
                spark.style.cssText = `position:absolute;left:${Math.random()*100}%;top:${Math.random()*100}%;width:6px;height:6px;background:hsl(${Math.random()*360},100%,70%);border-radius:50%;pointer-events:none;animation:fireworkAnim 2s ease-out forwards;z-index:100;`;
                fw.appendChild(spark); setTimeout(()=>spark.remove(),2000);
            }, i*80);
        }
        
        const style = document.createElement('style');
        style.textContent = `@keyframes fireworkAnim{0%{transform:scale(0) translateY(0);opacity:1;}100%{transform:scale(1) translateY(-100vh);opacity:0;}}`;
        document.head.appendChild(style);
        document.body.onclick = () => document.getElementById('welcomeScreen').style.display !== 'none' && revealSurprise();
    </script>
